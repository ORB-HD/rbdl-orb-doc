663  bool validate_spatial_axis (Math::SpatialVector &axis)
664  {
665 
666  bool axis_rotational = false;
667  bool axis_translational = false;
668 
669  Math::Vector3d rotation (axis[0], axis[1], axis[2]);
670  Math::Vector3d translation (axis[3], axis[4], axis[5]);
671 
672  if (fabs(rotation.norm()) > 1.0e-8) {
673  axis_rotational = true;
674  }
675 
676  if (fabs(translation.norm()) > 1.0e-8) {
677  axis_translational = true;
678  }
679 
680  if(axis_rotational && rotation.norm() - 1.0 > 1.0e-8) {
681  std::cerr << "Warning: joint rotation axis is not unit!" << std::endl;
682  }
683 
684  if(axis_translational && translation.norm() - 1.0 > 1.0e-8) {
685  std::cerr << "Warning: joint translation axis is not unit!" << std::endl;
686  }
687 
688  return axis_rotational != axis_translational;
689  }
